# 开始

```tsx
import { JsonView } from 'preact-json-view';
const arrayn = new Array(400).fill(1);
const json = {
  array: [1, 2, 3],
  array2: [
    [1, 2],
    [3, 4]
  ],
  arrayn,
  dateObj: new Date(),
  obj: {
    'first-child': 'hello',
    'second-child': 'world',
    'last-child': null,
  },
  str: 'hello world',
  strObj: new String('hello world'),
  numObj: new Number(10086),
  num: 1000,
  bigintObj: BigInt(10086),
  bigint: 10086n,
  boolObj: new Boolean(true),
  regExpObj: new RegExp('hello'),
  regExp: /hello/,
};
export default () => {
  return (
    <JsonView value={json} />
  )
}
```
